Ansible

Playbook


- hosts: webservers
  vars:
    http_port: 80
    max_clients: 200
  tasks:
  - name: ensure apache is at the latest version
    yum: name=httpd state=latest
  - name: ensure apache is running (and enable it at boot)
    service: name=httpd state=started enabled=yes
  handlers:
    - name: restart apache
      service: name=httpd state=restarted

SSH-nøkkel


- name: create ec2 key
  ec2_key:
    aws_access_key: "{{ aws_access_key }}"
    aws_secret_key: "{{ aws_secret_key }}"
    region: eu-central-1
    name: Min_nøkkel
    key_material: "{{ item }}"
  with_file: id_rsa.pub

Sikkerhetsgruppe


 - name: create load my super secure security group
   ec2_group:
     vpc_id: "{{ vpc.vpc_id }}"
     name: Min_sikkerhetsgruppe
     description: Åpner for HTTP-trafikk til lastbalanserer
     rules:
     - proto: tcp
       from_port: 80
       to_port: 80
       cidr_ip: 0.0.0.0/0
   register: load_balancer_group

Lastbalanserer


- name: configure load balancer
  ec2_elb_lb:
    security_group_names:
    - Min_sikkerhetsgruppe
    state: present
    name: Min_lastbalanserer
    health_check:
      ping_path: "/healthcheck.txt"
      # ...
    subnets:
    - "{{ vpc.subnets[0].id }}"
    - "{{ vpc.subnets[1].id }}"
    listeners:
    - protocol: http
      load_balancer_port: 80
      instance_port: 80

Jump server


- name: create jump server
  ec2:
    id: Min_unike_id
    key_name: Min_nøkkel
    instance_type: t2.micro
    image: ami-f9619996
    group: Min_ssh_sikkerhetsgruppe
    count: 1
    vpc_subnet_id: "{{ vpc.subnets[0].id }}"
    assign_public_ip: yes
    instance_tags:
      Name: Min_jump_server

Launch configuration


- name: configure launch configuration for autoscaling group
  ec2_lc:
    name: Min_launch_config
    image_id: ami-f9619996
    instance_type: t2.micro
    state: present
    user_data: "{{ lookup('file', 'startup.sh') }}"
    assign_public_ip: yes
    security_groups:
    - "{{ webservice_group.group_id }}"

Autoscaling group


- name: create autoscaling group
  ec2_asg:
    name: ansible-cloud-workshop-ag
    load_balancers:
    - Min_lastbalanserer
    availability_zones:
    - eu-central-1a
    - eu-central-1b
    launch_config_name: Min_launch_config
    min_size: 1
    max_size: 3
    desired_capacity: 2
    vpc_zone_identifier:
    - "{{ vpc.subnets[0].id }}"
    - "{{ vpc.subnets[1].id }}"
